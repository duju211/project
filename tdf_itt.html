stages_overview_raw <- function(stages_overview_url, stages_urls_css) {
  stages_overview_url_bow <- bow(stages_overview_url)
  
  stages_overview_html <- scrape(stages_overview_url_bow)
  
  stages_overview_nodes <- stages_overview_html |>
    html_nodes(stages_urls_css)
  
  tibble(
    href = html_attr(stages_overview_nodes, "href"),
    desc = html_text(stages_overview_nodes)) |>
    filter(str_detect(href, "stage-\\d+$")) |>
    distinct(href, .keep_all = TRUE)
}

Filter for time trial stages. Keep only rows containing the key word ‘\(ITT\)’ to do so. After filtering, further process the description to keep the string short and simple:

stages_itt <- function(df_stages_overview_combined_raw, time_trial_regex) {
  df_stages_overview_combined_raw |>
    filter(str_detect(desc, time_trial_regex)) |>
    mutate(
      url = str_glue("https://www.procyclingstats.com/{href}"),
      year = map_chr(str_split(year, "_"), ~ .x[length(.x)]),
      desc = str_remove(desc, time_trial_regex),
      desc = str_remove(desc, "Stage "),
      desc = str_remove(desc, "\\|"),
      desc = str_squish(desc),
      desc = str_glue("{desc} ({year})"))
}

stage_raw <- function(stage_url, stage_tbl_css) {
  stage_url_bow <- bow(stage_url)
  
  stage_html <- scrape(stage_url_bow)
  
  html_nodes(stage_html, stage_tbl_css) |>
    html_table() |>
    first() |>
    clean_names() |>
    mutate(url = stage_url)
}

Calculate time delta to winner time for each rider.

time_delta <- function(df_stage) {
  df_stage |>
    transmute(
      url, rider,
      time_delta = case_when(
        rnk == 1 ~ seconds(0),
        str_detect(time, ",") ~ ms(str_remove_all(time, ",")),
        TRUE ~ ms(str_sub(time, start = (str_length(time) / 2) + 1)))) |>
    filter(!is.na(time_delta))
}

Extract time for each winner:

winner_time <- function(df_stage) {
  df_stage |>
    group_by(url) |>
    summarise(winner_time = time[rnk == 1]) |>
    mutate(winner_time = hms(winner_time))
}

Analysis

Combine everything into one data frame. Calculate total time for each rider:

total_time <- function(df_time_delta, df_winner_time, df_stages_itt) {
  df_time_delta |>
    inner_join(df_winner_time, by = "url") |>
    transmute(
      url, rider,
      time = winner_time + time_delta,
      diff_winner_perc = (time - winner_time) / winner_time) |>
    left_join(df_stages_itt, by = "url")
}

Visualize data:

vis_total_time <- function(df_total_time) {
  df_total_time_pro <- df_total_time |>
    filter(diff_winner_perc != 0, diff_winner_perc != max(diff_winner_perc))
  
  df_summary <- df_total_time_pro |>
    group_by(desc) |>
    summarise(median_diff_winner_perc = median(diff_winner_perc)) |>
    arrange(desc(median_diff_winner_perc)) |>
    mutate(desc = fct_inorder(desc))
  
  df_vis <- df_total_time_pro |>
    mutate(desc = fct_rev(factor(desc, df_summary$desc)))
  
  df_vis |>
    ggplot(aes(y = desc, x = diff_winner_perc)) +
    ggbeeswarm::geom_beeswarm(alpha = 0.2) +
    geom_boxplot() +
    theme_light() +
    theme(axis.text.x = element_text(angle = 90)) +
    scale_x_continuous(labels = label_percent()) +
    labs(y = "Stage", x = "Difference to Winner Time [%]")
}
